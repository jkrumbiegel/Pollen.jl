function render!(io, x::Leaf, ::HTMLFormat)
reprs = x[].reprs
for mime in HTMLFormat_MIMES
if mime in keys(reprs)
print(io, adapthtmlstr(mime, reprs[mime]))
return
end
end
error(“Could not find mime for $(x[])!”)
end
